#Create a list of 5 numbers and print it
my_list=[20,21,22,23,24]
print(my_list)

#Add a new number to the end of a list using append()
a=[20,21,22,23,24]   # append means add the element
a.append(30)
print(a)

#Add multiple elements to a list using extend()
a.extend([31,32,33,34,35])  #add the bulk data
print(a)

#Insert a number at the second position in a list using insert().
a.insert(2,29)  # inserting the element
print(a)

#Remove a specific element from a list using remove()
a.remove(31)  #remove the element
print(a)

#Remove the last element from a list using pop()
a.pop(-1)  # it will remove the index
print(a)

#Find the length of a list using len()
print(len(a))

#Sort a list in ascending order using sort()
a.sort()
print(a)
#Reverse a list using reverse()
a.reverse()
print(a)
#Count how many times a given number appears in a list using count()
print(a.count(22))
