#1. Password Validation
#Scenario:
#Write a program that repeatedly asks the user to enter a password until the correct password "secret123" is entered. If the user enters the wrong password, print "Try again".
#Which loop structure will you use?
password = ""

while password != "secret123":
    password = input("Enter password: ")
    if password != "secret123":
        print("Try again")

print("Access granted")

#Sum of user inputs
total = 0

num = int(input("Enter a positive number (0 or negative to stop): "))

while num > 0:
    total += num
    num = int(input("Enter a positive number (0 or negative to stop): "))

print("Sum =", total)


#Day of the Week
num = int(input("Enter a number (1-7): "))#In Python, match-case does not need break

match num:
    case 1:
        print("Monday")
    case 2:
        print("Tuesday")
    case 3:
        print("Wednesday")
    case 4:
        print("Thursday")
    case 5:
        print("Friday")
    case 6:
        print("Saturday")
    case 7:
        print("Sunday")
    case 8:
        print("Invalid day")

#Operators
while True:
    op = input("Enter operator (+, -, *, /): ")
    a = int(input("Enter first number: "))
    b = int(input("Enter second number: "))

    match op:
        case "+":
            print("Result:", a + b)
            break
        case "-":
            print("Result:", a - b)
            break
        case "*":
            print("Result:", a * b)
            break
        case "/":
            if b == 0:
                print("Cannot divide by zero")
            else:
                print("Result:", a / b)
            break
        case 8:
            print("Invalid operator")
            break

#User roles and access
role = int(input("Enter role number (1-Admin, 2-Editor, 3-Viewer): "))

if role == 1:
    print("Access level: Admin (full access)")
elif role == 2:
    print("Access level: Editor (can edit content)")
elif role == 3:
    print("Access level: Viewer (read-only)")
else:
    print("Access denied")

#menu
while True:
    print("\n1. Add item")
    print("2. Delete item")
    print("3. View items")
    print("4. Exit")

    choice = int(input("Enter your choice (1-4): "))

    if choice == 1:
        item = input("Enter item to add: ")
        print(item, "added")

    elif choice == 2:
        item = input("Enter item to delete: ")
        print(item, "deleted")

    elif choice == 3:
        print("Viewing items")

    elif choice == 4:
        print("Exiting...")
        break

    else:
        print("Wrong choice")
