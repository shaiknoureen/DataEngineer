#1. Write a program to print numbers from 1 to 10 using a for loop.
for i in range(1,10):
    print(i)

#2. Write a program to print even numbers from 1 to 20.
for i in range(1,20):
    if i % 2 == 0:
        print("even number",i)

#3. Write a program to print odd numbers from 1 to 20.
for i in range(1,20):
    if i % 2 == 0:
        continue
    print("odd number",i)

#4. Write a program to calculate the sum of numbers from 1 to 10.
sum = 0
for i in range(1,10):
    sum = sum + i
print(sum)

#5. Write a program to print the multiplication table of a given number.
for i in range(1,11):
    a = 3
    mult = a * i
    print(mult)

#6. Write a program to count how many times a number appears in a list.
b = [1,2,1,3,4,1,5,6,1]
count = 0
for i in b:
    if i == 1:
        count = count + 1
print(count)

#7. Write a program to print each character of a string using a for loop.
name = "vamsi"
for character in name:
    print(character)

#8. Write a program to count vowels in a string.
name = "vamsi"
count = 0
for i in name:
    if i in "aeiouAEIOU":
        count = count + 1
print(count)
